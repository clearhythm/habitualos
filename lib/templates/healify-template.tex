% Healify Shift Cards LaTeX Template
% Uses Copernicus for H1 only and Poppins for everything else

\documentclass[$if(fontsize)$$fontsize$,$endif$$if(papersize)$$papersize$paper,$endif$]{article}

% Fonts
\usepackage{fontspec}
\setmainfont{$if(mainfont)$$mainfont$$else$Poppins$endif$}

% Geometry - includefoot to prevent footer cutoff
\usepackage[top=1in, bottom=0.5in, left=1in, right=1in, includefoot, footskip=0.5in]{geometry}

% Line spacing
\usepackage{setspace}
$if(linestretch)$
\setstretch{$linestretch$}
$endif$

% Colors - Healify branding
\usepackage{xcolor}
\definecolor{healifytitle}{RGB}{42, 74, 108}      % All headings
\definecolor{healifybody}{RGB}{0, 0, 0}           % Pure black for body text

% White background
\pagecolor{white}

% Graphics for images
\usepackage{graphicx}

% Adjust figure spacing - space above and below
\setlength{\intextsep}{24pt plus 2pt minus 2pt}  % Space above/below figures
\setlength{\abovecaptionskip}{4pt}
\setlength{\belowcaptionskip}{12pt}

% Set default text color for body
\color{healifybody}

% Custom footer
\usepackage{fancyhdr}

% Define light blue for footer background
\definecolor{footerblue}{RGB}{230, 241, 255}

% Configure footer
\pagestyle{fancy}
\fancyhf{} % Clear all header and footer fields
\fancyhead{} % Clear header
\renewcommand{\headrulewidth}{0pt} % Remove header line

% Custom footer with colored background - edge to edge
\fancyfoot[C]{%
  \hspace*{-1in}% Shift left by margin width
  \colorbox{footerblue}{%
    \makebox[\paperwidth]{%
      \parbox{0.9\paperwidth}{%
        \centering
        \vspace{18pt}
        \small
        Shift report generated by Healify @ healifyapp.com\\
        \textit{Mood tracking for a more radiant you}
        \vspace{18pt}
      }%
    }%
  }%
}

% Remove section numbering
\setcounter{secnumdepth}{0}

% Make headings use Healify styling
\usepackage{titlesec}

% H1 (\section) - Copernicus Trial font, 30% bigger
\titleformat{\section}
  {\fontspec{$if(headingfont)$$headingfont$$else$Copernicus Trial$endif$}\huge\bfseries\color{healifytitle}\linespread{1.0}\selectfont}{}{0em}{}
\titlespacing*{\section}{0pt}{0pt}{12pt}

% H2 (\subsection) - Poppins font, 30% bigger
\titleformat{\subsection}
  {\LARGE\bfseries\color{healifytitle}\linespread{1.0}\selectfont}{}{0em}{}

% H3 (\subsubsection) - Poppins font, 30% bigger
\titleformat{\subsubsection}
  {\large\bfseries\color{healifytitle}\linespread{1.0}\selectfont}{}{0em}{}

% Improved spacing
\usepackage{parskip}
\setlength{\parindent}{0pt}

% Pandoc tightlist command
\providecommand{\tightlist}{%
  \setlength{\itemsep}{0pt}\setlength{\parskip}{0pt}}

% Better quotes
\usepackage{csquotes}

% Hyperlinks (for any links in the document)
\usepackage{hyperref}
\hypersetup{
  colorlinks=true,
  linkcolor=healifytitle,
  urlcolor=healifytitle
}
